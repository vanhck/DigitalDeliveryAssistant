CREATE TABLE STACK (
  ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
  EMPFEANGER_ID INT NOT NULL ,
  ABLAGEORG_LAT FLOAT( 10, 6 ) NOT NULL ,
  ABLAGEORG_LNG FLOAT( 10, 6 ) NOT NULL ,
  PARKHINWEIS_LAT FLOAT( 10, 6 ) NOT NULL ,
  PARKHINWEIS_LNG FLOAT( 10, 6 ) NOT NULL ,
  STATUS VARCHAR(16),
  APPRAISED_ARRIVAL TIMESTAMP,
  ACTUAL_ARRIVAL TIMESTAMP
) ENGINE = MYISAM ;
